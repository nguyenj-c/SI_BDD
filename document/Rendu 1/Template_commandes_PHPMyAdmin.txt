CREATE DATABASE `horse_sim` COLLATE 'utf8_general_ci';

CREATE TABLE `horse_sim`.`club_hippique` ( `ID` INT(2) NOT NULL AUTO_INCREMENT , `ID_infra` INT(2) NOT NULL , `ID_concours` INT(2) NOT NULL ,
							`capacite` INT(2) NOT NULL ,  
							PRIMARY KEY (`ID`)) ENGINE = InnoDB;

CREATE TABLE `horse_sim`.`concours` ( `ID` INT(2) NOT NULL AUTO_INCREMENT , `ID_Item` INT(2) NOT NULL , `ID_infra` INT(2) NOT NULL ,
							`date_deb` TIMESTAMP NOT NULL , `date_fin` TIMESTAMP NOT NULL ,
							PRIMARY KEY (`ID`)) ENGINE = InnoDB;

CREATE TABLE `horse_sim`.`centre_equestre` ( `ID` INT(2) NOT NULL AUTO_INCREMENT , `ID_club` INT(2) NOT NULL , `ID_infra` INT(2) NOT NULL , `ID_taches_auto` INT(2) NOT NULL ,
							`capacite` INT(3) NOT NULL , 
							PRIMARY KEY (`ID`)) ENGINE = InnoDB;

CREATE TABLE `horse_sim`.`infrastructure` ( `ID` INT(2) NOT NULL AUTO_INCREMENT , `ID_accomodation_capacity` INT(2) NOT NULL , `ID_items` INT(2) NOT NULL , 
							`type` VARCHAR(10) NOT NULL , `niveau` INT(1) NOT NULL , `description` VARCHAR(50) NOT NULL , `infrastructure_family` VARCHAR(10) NOT NULL , `prix` INT(3) NOT NULL , `ressource_consumption` INT(3) NOT NULL , 
							PRIMARY KEY (`ID`)) ENGINE = InnoDB;

CREATE TABLE `horse_sim`.`joueur_compte` ( `ID_joueur_compte` INT(2) NOT NULL AUTO_INCREMENT , `ID_joueur` INT(2) NOT NULL , `ID_centre_equestre` INT(2) NOT NULL , `ID_chevaux` INT(2) NOT NULL , `ID_club_hippique` INT(2) NOT NULL , `ID_items` INT(2) NOT NULL , `ID_tache` INT(2) NOT NULL ,
							`username` INT(3) NOT NULL ,  `mdp` VARCHAR(50) NOT NULL , `date_Inscription` DATE NOT NULL , `last_connexion` DATETIME NOT NULL , 
							PRIMARY KEY (`ID_joueur_compte`,`ID_joueur`)) ENGINE = InnoDB;

CREATE TABLE `horse_sim`.`joueur` ( `ID_joueur` INT(2) NOT NULL AUTO_INCREMENT , 
							`nom` VARCHAR(20) NOT NULL , `prenom` VARCHAR(20) NOT NULL , `email` VARCHAR(30) NOT NULL , `sexe` ENUM('H','F') NOT NULL , `birthDate` DATE NOT NULL , `phone` INT(10) NOT NULL , `address` VARCHAR(50) NOT NULL ,  `IP_address` VARCHAR(15) NOT NULL , `profile_picture` VARCHAR(60) NOT NULL , `description` VARCHAR(50) NOT NULL , `website` VARCHAR(60) NOT NULL , 
							PRIMARY KEY (`ID_joueur`)) ENGINE = InnoDB;

CREATE TABLE `horse_sim`.`cheval` ( `ID` INT(2) NOT NULL AUTO_INCREMENT , `ID_attribut` INT(2) NOT NULL , `ID_etat` INT(2) NOT NULL ,
							`nom` VARCHAR(20) NOT NULL , `race` VARCHAR(10) NOT NULL , `description` VARCHAR(50) NOT NULL , 
							PRIMARY KEY (`ID`)) ENGINE = InnoDB;

CREATE TABLE `horse_sim`.`cheval_attribut` () ENGINE = InnoDB;

CREATE TABLE `horse_sim`.`etat` ( `ID` INT(2) NOT NULL AUTO_INCREMENT ,
							`type` ENUM('sante', 'moral', 'stress', 'fatigue', 'faim', 'proprete') NOT NULL ,
							PRIMARY KEY (`ID`)) ENGINE = InnoDB;

CREATE TABLE `horse_sim`.`banque_compte` ( `ID` INT(2) NOT NULL AUTO_INCREMENT , `ID_joueur` INT(2) NOT NULL ,
							`montant_compte` INT(5) NOT NULL , 
							PRIMARY KEY (`ID`)) ENGINE = InnoDB;

CREATE TABLE `horse_sim`.`historiquebanquaire` ( `ID` INT(2) NOT NULL AUTO_INCREMENT , `ID_banque_compte` INT(2) NOT NULL ,
							`operation_effetuee` VARCHAR(10) NOT NULL , `montant` INT(5) NOT NULL , `last_connexion` DATETIME NOT NULL , 
							PRIMARY KEY (`ID`)) ENGINE = InnoDB;

CREATE TABLE `horse_sim`.`tache_auto` ( `ID` INT(2) NOT NULL AUTO_INCREMENT ,
							`action` VARCHAR(30) NOT NULL , `frequence` VARCHAR(10) NOT NULL , 
							PRIMARY KEY (`ID`)) ENGINE = InnoDB;

CREATE TABLE `horse_sim`.`journal` ( `ID` INT(2) NOT NULL AUTO_INCREMENT , `ID_joueur` INT(2) NOT NULL , `ID_article` INT(2) NOT NULL , 
							`date` TIMESTAMP NOT NULL , `agenda` VARCHAR(45) NOT NULL , `top_events` VARCHAR(45) NOT NULL , 
							PRIMARY KEY (`ID`)) ENGINE = InnoDB;

CREATE TABLE `horse_sim`.`article` ( `ID` INT(2) NOT NULL AUTO_INCREMENT , 
							`title` VARCHAR(20) NOT NULL , `text` VARCHAR(50) NOT NULL , `image` VARCHAR(40) NOT NULL , `category` VARCHAR(10) NOT NULL , 
							PRIMARY KEY (`ID`)) ENGINE = InnoDB;

CREATE TABLE `horse_sim`.`item` ( `ID` INT(2) NOT NULL AUTO_INCREMENT , `ID_niveau` INT(2) NOT NULL , `ID_famille` INT(2) NOT NULL ,
							`type` VARCHAR(20) NOT NULL , `description` VARCHAR(50) NOT NULL , `prix` INT(3) NOT NULL ,
							PRIMARY KEY (`ID`)) ENGINE = InnoDB;

CREATE TABLE `horse_sim`.`famille_item` ( `ID` INT(2) NOT NULL AUTO_INCREMENT , 
							`nom` VARCHAR(20) NOT NULL , 
							PRIMARY KEY (`ID`)) ENGINE = InnoDB;

CREATE TABLE `horse_sim`.`niveau` ( `ID` INT(2) NOT NULL AUTO_INCREMENT , 
							`rarete` VARCHAR(10) NOT NULL ,  
							PRIMARY KEY (`ID`)) ENGINE = InnoDB;

CREATE TABLE `horse_sim`.`Magasin` ( `ID` INT(2) NOT NULL AUTO_INCREMENT , `ID_centre_equestre` INT(2) NOT NULL , `ID_club` INT(2) NOT NULL , `ID_infra` INT(2) NOT NULL , `ID_item` INT(2) NOT NULL ,
							`nom` VARCHAR(20) NOT NULL ,
							PRIMARY KEY (`ID`)) ENGINE = InnoDB;
